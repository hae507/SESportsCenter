CREATE TABLE `Member` (
	`id`	VARCHAR(20)	NOT NULL,
	`name`	VARCHAR(10)	NOT NULL,
	`password`	VARCHAR(20)	NOT NULL,
	`phoneNum`	VARCHAR(15)	NOT NULL
);

CREATE TABLE `Instructor` (
	`instructor_id`	VARCHAR(20)	NOT NULL,
	`instructor_pw`	VARCHAR(20)	NOT NULL,
	`instructor_name`	VARCHAR(20)	NOT NULL,
	`instructor_phone`	VARCHAR(15)	NOT NULL,
	`instructor_ssn`	VARCHAR(20)	NOT NULL,
	`instructor_add`	VARCHAR(50)	NOT NULL
);

CREATE TABLE `Lesson` (
	`LessonId`	VARCHAR(20)	NOT NULL,
	`instructor_id`	VARCHAR(20)	NOT NULL,
	`LessonName`	VARCHAR(30)	NOT NULL,
	`LessonTime`	TIME	NOT NULL,
	`LessonStartTime`	TIME	NOT NULL,
	`LessonEndTime`	TIME	NOT NULL,
	`LessonAvailDate`	DATE	NOT NULL,
	`LessonPrice`	INT	NOT NULL,
	`LessonDCRate`	INT	NOT NULL,
	`SportsCenterName`	VARCHAR(20)	NOT NULL
);

CREATE TABLE `Locker` (
	`lockerId`	VARCHAR(20)	NOT NULL,
	`lockerLocation`	VARCHAR(30)	NOT NULL,
	`rentalAmount`	DOUBLE	NOT NULL,
	`rentalPeriod`	DATE	NOT NULL,
	`centerName`	VARCHAR(20)	NOT NULL
);

CREATE TABLE `Sales` (
	`SalesNum`	VARCHAR(20)	NOT NULL,
	`enrollLessonNum`	INT	NULL,
	`EnrollLockerNum`	INT	NULL,
	`amount`	INT	NOT NULL,
	`date`	DATE	NOT NULL,
	`center`	VARCHAR(20)	NOT NULL
);

CREATE TABLE `Notice` (
	`notice_no`	INT	NOT NULL,
	`notice_title`	VARCHAR(50)	NOT NULL,
	`notice_contents`	VARCHAR(500)	NOT NULL,
	`notice_updateTime`	DATETIME	NOT NULL
);

CREATE TABLE `EnrollLesson` (
	`enrollLessonNum`	INT	NOT NULL,
	`id`	VARCHAR(20)	NOT NULL,
	`LessonId`	VARCHAR(20)	NOT NULL
);

CREATE TABLE `EnrollLocker` (
	`EnrollLockerNum`	INT	NOT NULL,
	`id`	VARCHAR(20)	NOT NULL,
	`lockerId`	VARCHAR(20)	NOT NULL
);
